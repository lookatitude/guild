## Shared invariants

Every skill file authored in P1 must pass:

```python
import re, json, sys
p = sys.argv[1]; s = open(p).read()
m = re.match(r"^---\n(.*?)\n---", s, re.DOTALL); assert m, "no frontmatter"
block = m.group(1)
assert re.search(r"^name:\s*\S", block, re.M), "missing name"
dm = re.search(r"^description:\s*(.+?)(?=^\w+:|\Z)", block, re.M | re.S); assert dm, "missing description"
desc = dm.group(1).strip()
assert len(desc) <= 1024, f"description {len(desc)} > 1024"
assert re.search(r"^when_to_use:\s*\S", block, re.M), "missing when_to_use"
# evals.json for this skill
evals_path = p.rsplit("/",1)[0] + "/evals.json"
with open(evals_path) as f: ev = json.load(f)
assert len(ev.get("should_trigger", [])) >= 3, "need ≥3 should_trigger"
assert len(ev.get("should_not_trigger", [])) >= 3, "need ≥3 should_not_trigger"
print(f"OK {p}")
```
